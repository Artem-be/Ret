Функция ВойтиНаСервере()
	 ПользовательНайден = Справочники.Пользователи.НайтиПоНаименованию(Логин, Истина);
    Если ПользовательНайден.Пустая() Тогда
        Сообщить("Вы ввели неверный логин или пароль. Пожалуйста, проверьте ещё раз введенные данные");
        Возврат Неопределено;
    КонецЕсли;
    Если ПопыткиПользователя(ПользовательНайден) >= 3 Тогда
        ИзменитьДанныеНаСервере(ПользовательНайден, "Активность", Ложь);
    КонецЕсли;
    Если НЕ АктивностьПользователя(ПользовательНайден) Тогда
        Сообщить("Вы заблокированы. Обратитесь к администратору");
        Возврат Неопределено;
    КонецЕсли;
    Если ПользовательНайден.Пароль <> Пароль Тогда
        УстановитьКоличествоПопыток(ПользовательНайден);
        Сообщить("Вы ввели неверный логин или пароль. Пожалуйста, проверьте ещё раз введенные данные");
        Возврат Неопределено;
    КонецЕсли;
    ПоследнийВход = ВходПользователя(ПользовательНайден);
    Если ПоследнийВход <> Дата('00010101') Тогда
        Если ТекущаяДата() > ДобавитьМесяц(ПоследнийВход, 1) Тогда
            ИзменитьДанныеНаСервере(ПользовательНайден, "Активность", Ложь);
            Сообщить("Вы заблокированы. Обратитесь к администратору");
            Возврат Неопределено;
        КонецЕсли;
    КонецЕсли;
    Возврат ПользовательНайден.Роль;;
КонецФункции

&НаКлиенте
Процедура Войти(Команда)
	РольПользователя = ВойтиНаСервере();
    Если ЗначениеЗаполнено(РольПользователя) Тогда
        ИзменитьДанныеНаСервере(ПользовательНайден, "КоличествоПопыток", 0);
        Если ВходПользователя(ПользовательНайден) = Дата('00010101') Тогда
            Сообщить("Измените пароль при первом входе");
            ПараметрыФормы = Новый Структура("ПараметрПользователь", ПользовательНайден);
            ОткрытьФорму("ОбщаяФорма.ФормаСменыПароля", ПараметрыФормы);
        Иначе
            ИзменитьДанныеНаСервере(ПользовательНайден, "ДатаВхода", ТекущаяДата());
            Сообщить("Вы успешно авторизовались");
            Если РольПользователя = ПредопределенноеЗначение("Перечисление.Роли.Администратор") Тогда
                ОткрытьФорму("Справочник.Пользователи.ФормаСписка");
            ИначеЕсли РольПользователя = ПредопределенноеЗначение("Перечисление.Роли.Пользователь") Тогда
                ОткрытьФорму("ОбщаяФорма.ФормаПользователя");
            КонецЕсли;
        КонецЕсли;
        ЭтаФорма.Закрыть();
    КонецЕсли;
