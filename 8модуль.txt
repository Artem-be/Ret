Процедура ПередЗаписью(Отказ)

	Если ОбменДанными.Загрузка Тогда
		Возврат;
	КонецЕсли;

	ТекущийЛогин = ЭтотОбъект.Наименование;
	ТекущаяСсылка = ЭтотОбъект.Ссылка;

	Запрос = Новый Запрос;
	Запрос.Текст = 
		"ВЫБРАТЬ
		|	Пользователи.Ссылка КАК Ссылка
		|ИЗ
		|	Справочник.Пользователи КАК Пользователи
		|ГДЕ
		|	Пользователи.Наименование = &Логин
		|	И Пользователи.Ссылка <> &ТекущаяСсылка";

	Запрос.УстановитьПараметр("Логин", ЭтотОбъект.Наименование);
	Запрос.УстановитьПараметр("ТекущаяСсылка", ТекущаяСсылка);

	Выборка = Запрос.Выполнить().Выбрать();

	Если Выборка.Количество() > 0 Тогда
		Отказ = Истина;
		Сообщить("Логин """ + ТекущийЛогин + """ уже используется другим пользователем!" + Символы.ПС + "Выберите другой логин.");
	КонецЕсли;

КонецПроцедуры
